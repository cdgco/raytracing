69  static Vector3D Color(const Ray &r, vList &vector, int iDepth) {
70  HitRecord temp_rec, rec;
71  bool bHitAnything = false;
72  double dClosestSoFar = DBL_MAX;
73  for (size_t i = 0; i < vector.size(); i++) {
74  if (vector[i]->Hit(r, temp_rec, 0.001, dClosestSoFar)) {
75  bHitAnything = true;
76  dClosestSoFar = temp_rec.m_dT;
77  rec = temp_rec;
78  }
79  }
80  if (bHitAnything) {
81  Ray m_scattered;

83  if (iDepth < 50 && rec.m_pmCurMat->Scatter(r, rec, m_attenuation, m_scattered)) {
84  return m_attenuation * Color(m_scattered, vector, iDepth + 1);
85  }
86  else {
87  return Vector3D(0);
88  }
89  }
90  else {

126  double u = double(i + drand48()) / double(m_Dims.iX); // Randomize location of ray within pixel (x)

128  Ray m_r = m_camera.GetRay(u, v); // Create ray from randomized location
129 
130  m_col += Color(m_r, m_list, 0); // Sum all anti-aliased rays
131  }
132  m_col /= double(m_iRaysPerPixel); // Get average color from all samples taken (anti-aliasing)
133  m_col = Vector3D(sqrt(m_col[0]), sqrt(m_col[1]), sqrt(m_col[2])); // Correct gamma of pixel
134 
135  // Convert pixel color values to 8-bit color depth (0-255)
136  int iRed = int(255.99*m_col[0]);
