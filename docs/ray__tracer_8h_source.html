97  HitRecord temp_rec, rec;

99  double dClosestSoFar = DBL_MAX;
100  for (size_t i = 0; i < vector.size(); i++) {
101  if (vector[i]->Hit(r, temp_rec, 0.001, dClosestSoFar)) {

151  for (int s = 0; s < m_iRaysPerPixel; s++) { // For each anti-aliasing sample
152  double u = double(i + drand48()) / double(m_dims.m_iX); // Randomize location of ray within pixel (x)
153  double v = double(j + drand48()) / double(m_dims.m_iY); // Randomize location of ray within pixel (y)
154  Ray m_r = m_camera.GetRay(u, v); // Create ray from randomized location
155 
156  m_col += Color(m_r, m_list, 0); // Sum all anti-aliased rays
157  }
158  m_col /= double(m_iRaysPerPixel); // Get average color from all samples taken (anti-aliasing)
159  m_col = Vector3D(sqrt(m_col[0]), sqrt(m_col[1]), sqrt(m_col[2])); // Correct gamma of pixel
160 
161  // Convert pixel color values to 8-bit color depth (0-255)
162  int iRed = int(255.99*m_col[0]);
